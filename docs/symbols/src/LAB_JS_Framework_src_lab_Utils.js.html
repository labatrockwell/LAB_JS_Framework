 54  * example: aNumber.x.sign();
 55  */
 56 Number.prototype.sign = function() {
 57   if(this > 0) {
 58     return 1;
 59   } else if (this < 0) {
 60     return -1;
 61   } else {
 62     return 0;
 63   }

112 	// get string from url. e.g. www.lab.com/index.html?sandwich=turkey returns 'turkey'
113 	/**
114 	 @function
115 	 */
116 	LAB.getQueryString = function (key)
117 	{
118 		key = key.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");
119 		var regex = new RegExp("[\\?&]"+key+"=([^&#]*)");
120 		var qs = regex.exec(window.location.href);
121 		if(qs == null)
122 			return '';
123 		else
124 			return qs[1];
125 	}

137 	// add bind method for browsers that don't currently support it (such as Safari)
138 	if (!Function.prototype.bind) {  
139 	  Function.prototype.bind = function (oThis) {  
140 		if (typeof this !== "function") {  
141 		  // closest thing possible to the ECMAScript 5 internal IsCallable function  
142 		  throw new TypeError("Function.prototype.bind - what is trying to be bound is not callable");  
143 		}  
144 	  
145 		var aArgs = Array.prototype.slice.call(arguments, 1),   
146 			fToBind = this,   
147 			fNOP = function () {},  
148 			fBound = function () {  
149 			  return fToBind.apply(this instanceof fNOP  
150 									 ? this  
151 									 : oThis || window,  
152 								   aArgs.concat(Array.prototype.slice.call(arguments)));  
153 			};  
154 	  
155 		fNOP.prototype = this.prototype;  
156 		fBound.prototype = new fNOP();  
157 	  
158 		return fBound;  
159 	  };  
160 	}

163 /********************************************
164 	NAMESPACE
165 ********************************************/
166 /**
167  * Use this function to safely create a new namespace
168  * if a namespace already exists, it won't be recreated.
169  *
170  * @function
171  * @param {String} nsString The namespace as a string.
172  * @return {Object} The namespace object.
173  */
174 window.namespace = function (namespaceString) {
175 	var parts = namespaceString.split('.'),
176 		parent = window,
177 		i;
178 
179 	for (i=0; i<parts.length; i +=1) {
180 		// create a property if it doesn't exist
181 		if (typeof parent[parts[i]] === "undefined") {
182 			parent[parts[i]] = {};
183 		}
184 		parent = parent[parts[i]];
185 	}
186 	return parent;
187 };
