 52 /**
 53  * @returns The sign of this number, 0 if the number is 0.
 54  * example: aNumber.x.sign();
 55  */
 56 Number.prototype.sign = function() {
 57   if(this > 0) {
 58     return 1;
 59   } else if (this < 0) {
 60     return -1;
 61   } else {
 62     return 0;
 63   }
 64 }

112 	// get string from url. e.g. www.lab.com/index.html?sandwich=turkey returns 'turkey'
113 	/**
114 	 @function
115 	 */
116 	LAB.getQueryString = function (key)
117 	{
118 		key = key.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");
119 		var regex = new RegExp("[\\?&]"+key+"=([^&#]*)");
120 		var qs = regex.exec(window.location.href);
121 		if(qs == null)
122 			return '';
123 		else
124 			return qs[1];
125 	}
