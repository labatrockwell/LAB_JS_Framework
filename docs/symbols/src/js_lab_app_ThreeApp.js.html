 37 /************************************************************

 40 
 41 	/** 
 42 		Call to set up Webgl, initialize the canvas, provide default THREE vars
 43 		and start animate() loop
 44 		@function 
 45 		@public
 46 	*/
 47 
 48 	LAB.app.ThreeApp.prototype.begin = function()
 49 	{
 50 		/**
 51 		* default THREE camera
 52 		* @type THREE.Camera
 53 		*/
 54 		console.log("base app set up");
 55 		this.camera = new THREE.Camera( 70, window.innerWidth / window.innerHeight, 1, 10000 );
 56 		this.camera.position.y = 300;
 57 		this.camera.position.z = 500;
 58 		
 59 		/**
 60 		* default THREE scene
 61 		* @type THREE.Scene
 62 		*/
 63 		this.scene = new THREE.Scene();
 64 		
 65 		/**
 66 		* default THREE projector
 67 		* @type THREE.Projector
 68 		*/
 69 		this.projector = new THREE.Projector();
 70 		
 71 		/**
 72 		* default THREE renderer with anti-aliasing, depth sorting off
 73 		* @type THREE.WebGLRenderer
 74 		*/
 75 		this.renderer = new THREE.WebGLRenderer( { antialias: true } );
 76 		this.renderer.sortObjects = false;
 77 		this.renderer.setSize( window.innerWidth, window.innerHeight );
 78       this.renderer.autoClear = false;
 79 
 80 		// do we have a container?
 81 	
 82 		if (document.getElementById("labContainer") != null){
 83 			this.container = document.getElementById("labContainer");
 84 		} else {
 85 			console.log("no labContainer in document, generating container div")
 86 			this.container = document.createElement( 'div' );
 87 			if (document.body)
 88 				document.body.appendChild( this.container );
 89 			else
 90 				return;
 91 		}
 92 		
 93 		this.container.appendChild(this.renderer.domElement);	
 94 		
 95 		gl = gl || this.renderer.getContext();
 96 		
 97 		this.setup();
 98 		this.animate();
 99 	}
