 49 	/** 
 50 		Call to set up Webgl, initialize the canvas, provide default THREE vars
 51 		and start animate() loop
 52 		@function 
 53 		@public
 54 		@param width (optional) width of renderer
 55 		@param height (optional) height of renderer
 56 	*/
 57 
 58 	LAB.app.ThreeApp.prototype.begin = function(parameters)
 59 	{
 60 		parameters = parameters || {};
 61 		/**
 62 		* default THREE camera
 63 		* @type THREE.Camera
 64 		*/
 65 		console.log("base app set up");
 66 		
 67 		// listen to mouse + keys + window by default
 68 		this.registerWindowEvents();
 69 		this.registerKeyEvents();
 70 		this.registerMouseEvents();
 71 			
 72 		/**
 73 		* default THREE renderer with anti-aliasing, depth sorting off
 74 		* @type THREE.WebGLRenderer
 75 		*/
 76 		this._width = parameters.width || window.innerWidth;
 77 		this._height = parameters.height || window.innerHeight;
 78 
 79 		// create canvas so we can customize it a bit
 80 		this._canvas = parameters.canvas !== undefined ? parameters.canvas : document.createElement( 'canvas' );
 81 		this._canvas.id = parameters.canvasId !== undefined ? parameters.canvasId : "labCanvas";
 82 
 83 		this.renderer = new THREE.WebGLRenderer( { antialias: (parameters.antialias !== undefined ? parameters.antialias : true), canvas:this._canvas } );
 84 		this.renderer.sortObjects = false;
 85 		this.renderer.setSize( this._width, this._height );
 86       	//this.renderer.autoClear = false;
 87 
 88 		this.camera = new THREE.PerspectiveCamera(60, this._width / this._height, .1, 2000);
 89 	    this.camera.position.set( 0, 0, 50 );
 90 	    this.camera.lookAt( new THREE.Vector3(0, 0, 0) );
 91 
 92 		//this.camera = new LAB.three.Camera( 60, this._width / this._height, .1, 2000 );
 93         //this.camera.setPerspective( 60, this._width, this._height );
 94         //this.camera.usePushPop( true );
 95 		
 96 		/**
 97 		* default THREE scene
 98 		* @type THREE.Scene
 99 		*/
100 		this.scene = parameters.scene || new THREE.Scene();
101 		this.scene.add(this.camera);
102       			
103 		/**
104 		* default THREE projector
105 		* @type THREE.Projector
106 		*/
107 		this.projector = new THREE.Projector();
108       	
109 		// do we have a container?
110 	
111 		if (document.getElementById("labContainer") != null){
112 			this.container = document.getElementById("labContainer");
113 		} else {
114 			console.log("no labContainer in document, generating container div")
115 			this.container = document.createElement( 'div' );
116 			if (document.body){
117 				document.body.appendChild( this.container );
118 			} else {
119  				return;
120 			}
121 		}
122 		
123 		this.container.appendChild(this.renderer.domElement);	
124 		
125 		gl = gl || this.renderer.getContext();
126 		
127 		this.setup();
128 		this.animate();
129 	}
