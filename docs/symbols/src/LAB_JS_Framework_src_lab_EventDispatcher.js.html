1 /** 
  2  * 
  3  * @namespace LAB 
  4  * @constructor
  5  * param {Class} target The instance of the class that implements the EventDispatcher
  6  */
  7 LAB.EventDispatcher = function(target)
  8 {
  9 	var _target = target || null;
 10 	this.getTarget = function() {
 11 		return _target;
 12 	}
 13 	this.events = [];
 14 }
 15 
 16 LAB.EventDispatcher.prototype = {
 17 	constructor:LAB.EventDispatcher,
 18 	/**
 19 	 * @param {String} event The event type
 20 	 * @param {Function} callback The function to be called when the event is dispatched
 21 	 */
 22 	addEventListener: function(type, callback){
 23 		this.events[type] = this.events[type] || []; // check for undefined arg
 24 		if (this.events[type]){
 25 			this.events[type].push(callback);
 26 		}
 27 	},
 28 	/**
 29 	 * @param {String} event The event type
 30 	 * @param {Function} callback The function to be called when the event is dispatched
 31 	 * return {boolean} True if is successful, false if not.
 32 	 */
 33 	removeEventListener: function(type,callback){
 34 		if ( this.events[type] ) {
 35 			var listeners = this.events[type];
 36 			for ( var i = listeners.length-1; i>=0; --i ){
 37 				if ( listeners[i] === callback ) {
 38 					listeners.splice( i, 1 );
 39 					return true;
 40 				}
 41 			}
 42 		}
 43 		return false
 44 	},
 45 	/**
 46 	 * @param {String} event The event type
 47 	 * return {boolean} True if listener exists for thie event type, false if not
 48 	 */
 49 	hasEventListener:function(type) {
 50 		if (this.events[type]) {
 51 			return true;
 52 		} else {
 53 			return false;
 54 		}
 55 	},
 56 	/**
 57 	 * There are 2 ways to use dispatchEvent. The first is to send an Event object (including
 58 	 * objects that extend the Event class) and any optionl parameters. For example:
 59 	 *
 60 	 * @example
 61 	 * someObject.dispatchEvent(new LAB.Event(Event.CHANGE));
 62 	 * @example
 63 	 * var params = {secret: "my secret", message: "hi"};
 64 	 * someObject.dispatchEvent(new LAB.Event("customEventType"), params);
 65 	 * <p>The second is to pass a string as the event type and any optional params</p>
 66 	 * @example
 67 	 * someObject.dispatchEvent("customEventType");
 68 	 * @example
 69 	 * var params = {secret: "my secret", message: "hi"};
 70 	 * someObject.dispatchEvent("customEventType", params);	 
 71 	 *
 72 	 * @param {string|Event} typeOrEvent The event type as a string or an Event object
 73 	 * @param {Object} optionalParams Optional parameters passed as an object.
 74 	 * return {boolean} True if dispatch is successful, false if not.
 75 	 */
 76 	dispatchEvent:function(typeOrEvent, optionalParams){
 77 		
 78 		if (typeof typeOrEvent === 'string') {
 79 			var event = new LAB.Event(typeOrEvent);
 80 		} else if (typeof typeOrEvent=== 'object') {
 81 			var event = typeOrEvent;
 82 		} else {
 83 			throw "First parameter of dispatchEvent must be of type object or string";
 84 		}
 85 		
 86 		event.target = this.getTarget();
 87 		
 88 		for (var obj in optionalParams) {
 89 			event[obj.toString()] = optionalParams[obj];
 90 		}
 91 		
 92 		if ( this.events[event.type] ) {
 93 			var listeners = this.events[event.type], len = listeners.length;
 94 			while ( len-- ) {
 95 				listeners[len](event);	//callback with self
 96 			}		
 97 			return true; // return success
 98 		} else {
 99 			return false;
100 		}
101 	}
102 }

104 /** 
105  * Event 'base class' (but it can also be instantiated directly)
106  *
107  * @constructor
108  * @param {String} type event type
109  */
110 LAB.Event = function(type) {
111 	this.type = type;
112 	this.target = null;
113 }
