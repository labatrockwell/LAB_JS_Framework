  9 	@extends LAB.app.BaseApp
 10 	@extends TSPS.dispatcher

 12 LAB.app.TSPSApp = function()
 13 {
 14 	TSPS.Dispatcher.call( this );
 15 	LAB.app.BaseApp.call( this );
 16 	
 17 	this.socket 	= null;
 18 }

 26 /**
 27 Connect to TSPS WebSocket
 28 @function
 29 */
 30 LAB.app.TSPSApp.prototype.connect = function (port, host)
 31 {
 32 	host = host || "ws://localhost";
 33 	
 34 	this.socket = new LAB.utils.WebSocket( host+":"+port, "tsps-protocol");
 35 	this.socket._onMessageReceived	= this._onMessageReceived.bind(this);
 36 	this.socket.onMessageReceived 	= this.onMessageReceieved;
 37 	this.socket.onConnectionOpened 	= this.onConnectionOpened;
 38 	this.socket.onConnectionClosed 	= this.onConnectionClosed;
 39 	this.socket.connect();
 40 	
 41 	console.log("connecting to "+ host+":"+port);
 42 }
