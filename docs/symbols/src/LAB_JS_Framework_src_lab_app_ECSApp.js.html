  8 /** @constructor */
  9 	LAB.app.ECSApp = function()
 10 	{
 11 		this._api				= null;

 13 		this.bECSSetup			= false;
 14 		this.bPerformanceStarted	= false;
 15 	}

 17 /** 
 18 	setup: call this from your main app
 19 	@function 
 20 	@public
 21 */
 22 
 23 	LAB.app.ECSApp.prototype.setupECS 		= 	function(uuid, inport, outport)
 24 	{	
 25 		this.uuid			= uuid;
 26 		this.ecs_in_port 	= inport;
 27 		this.ecs_out_port 	= outport;
 28 
 29 		this._api = new ApplicationMessageHandler(uuid);
 30 
 31 		this._api.addEventListener("handleStartPerformance", this.handleStartPerformance);
 32 		this._api.addEventListener("handleStopPerformance", this.handleStopPerformance);
 33 		this._api.addEventListener("handleConfig", this.handleConfig);
 34 		this._api.addEventListener("handlePerformanceConfig", this.handlePerformanceConfig);
 35 				
 36 		this._api.connect(this.ecs_in_port, this.ecs_out_port);
 37 		this.bECSSetup = true;	
 38 	};
