 14 LAB.sound.WAPlayer = function( ){//loop, autoplay, controls ){		
 15 	LAB.EventDispatcher.call( this, this );
 16 	
 17 	this.buffer = null;
 18 	this.pan	= new LAB.geom.Point();
 19 	this.pitch	= 0;
 20 	this.gain	= 0;
 21 	
 22 	this.filter = null;
 23 	
 24 	this.currentSource 	= null;
 25 	this.currentTime	= 0;
 26 	
 27 	this.panner			= LAB.sound.Context.createPanner();
 28 	this.panner.setPosition(0,0,0);
 29 	
 30 	this.start 	= 0;
 31 	this.end	= 0;
 32 }

 36 LAB.sound.WAPlayer.prototype.loadFile 		= function( url )
 37 {	
 38 	//this.registerEvents();
 39 	var request = new XMLHttpRequest();
 40 	request.parent = this;
 41     request.open("GET", url, false);
 42     request.responseType = "arraybuffer";
 43 	request.parent = this;
 44 	
 45     request.onload = function() { 
 46         request.parent.buffer = LAB.sound.Context.createBuffer(request.response, true);
 47 		request.parent.end = Math.floor(request.parent.buffer.duration);
 48     }
 49 
 50     request.send();	
 51 }

 60 LAB.sound.WAPlayer.prototype.play	= function(){
 61 	//this.source.noteOff(0);
 62 	var source 					= LAB.sound.Context.createBufferSource();	
 63 	source.playbackRate.value 	= this.pitch;
 64 	source.gain					= this.gain;
 65 	source.buffer 				= this.buffer;
 66 	
 67 	if (this.filter != null){
 68 		source.connect(this.filter);
 69 		this.filter.connect(this.panner);
 70 	}  else {
 71 		source.connect(this.panner);
 72 	}
 73 	
 74 	this.panner.connect(LAB.sound.Context.destination);
 75 	source.noteGrainOn(0, this.start, this.end);
 76 }

 82 LAB.sound.WAPlayer.prototype.setPan	= function(x, y, z){
 83 	this.pan.x = x;
 84 	this.pan.y = y || this.pan.y;
 85 	this.pan.z = z || this.pan.z;
 86 	this.panner.setPosition(this.pan.x, this.pan.y, this.pan.z);
 87 }
