  5 /** 
  6 	@constructor 
  7 	@param id without parent: unique ID of html element. with parent: id of child of parent div
  8 	@param parent (optional) either LAB.utils.DOMElement or string
  9 */
 10 LAB.utils.DOMElement = function( id, parent ){
 11 	if (id){
 12 		if (parent){
 13 			if (typeof(parent) == "string"){
 14 				this.parent = document.getElementById(parent);	
 15 				if ( this.parent != null ){
 16 					this.element = this.parent.querySelector( "#"+id );
 17 				} else {
 18 					console.warn("parent is null!")
 19 				}
 20 			// assuming you have passed a LAB.utils.DOMElement
 21 			} else {
 22 				this.parent = parent;
 23 				this.element = this.parent.element.querySelector( "#"+id );
 24 			}
 25 		} else {
 26 			this.element = document.getElementById(id);	
 27 			if ( this.element == null ){
 28 				this.element = document.createElement("div");
 29 				console.warn("invalid id, this.element = div");
 30 			}
 31 		}		
 32 	}
 33 }

 88 LAB.utils.DOMElement.prototype.hide = function(){
 89 	this.element.style.visibility = "hidden";
 90 	//this.element.style.display 	= "none";
 91 }
 92 
 93 LAB.utils.DOMElement.prototype.show = function(){
 94 	this.element.style.visibility = "visible";
 95 	//this.element.style.display	  = "box";
 96 }
 97 
 98 LAB.utils.DOMElement.prototype.update = function(){
 99 	if ( this.opacity == "0" && !this.hidden ){
100 		this.hide();
101 	}
102 }
