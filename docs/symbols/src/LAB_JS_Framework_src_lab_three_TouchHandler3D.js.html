  5 /**

 10 LAB.three.TouchHandler3D = function(scene, camera) {
 11 	"use strict";
 12 	
 13 	LAB.EventDispatcher.call(this, this);
 14 	
 15 	var scene = scene;
 16 	var camera = camera;
 17 	var self = this;
 18 
 19 	var hash = {};
 20 	
 21 	var projector;
 22 	
 23 	var gestureHandler;
 24 	
 25 	projector = new THREE.Projector();
 26 	
 27 	gestureHandler = new LAB.TouchGestureHandler();
 28 	gestureHandler.register(window);	
 29 	
 30 	/**
 31 	 * @private
 32 	 */
 33 	function addGestureEventListener(type) {
 34 	
 35 		// map 3d touch events to 2d touch events
 36 		var eventMap = {
 37 						"touch3DDown": LAB.TouchEvent.PRESS,
 38 						"touch3DUp": LAB.TouchEvent.RELEASE,
 39 						"touch3DFlick": LAB.TouchEvent.FLICK,
 40 						"touch3DDrag": LAB.TouchEvent.DRAG,
 41 						"touch3DTap": LAB.TouchEvent.TAP,
 42 						"touch3DOut": LAB.TouchEvent.OUT
 43 						};
 44 		
 45 		if (!gestureHandler.hasEventListener(eventMap[type])) {
 46 		
 47 			switch (type) {
 48 				case LAB.three.TouchEvent3D.DOWN:
 49 					gestureHandler.addEventListener(LAB.TouchEvent.PRESS, onDownEvent);
 50 					break;
 51 				case LAB.three.TouchEvent3D.UP:
 52 					gestureHandler.addEventListener(LAB.TouchEvent.RELEASE, onUpEvent);
 53 					break;
 54 				case LAB.three.TouchEvent3D.FLICK:
 55 					gestureHandler.addEventListener(LAB.TouchEvent.FLICK, onTouchFlick);
 56 					break;
 57 				case LAB.three.TouchEvent3D.DRAG:
 58 					gestureHandler.addEventListener(LAB.TouchEvent.DRAG, onTouchDrag);
 59 					break;
 60 				case LAB.three.TouchEvent3D.TAP:
 61 					gestureHandler.addEventListener(LAB.TouchEvent.TAP, onTouchTap);
 62 					break;
 63 				case LAB.three.TouchEvent3D.OUT:
 64 					gestureHandler.addEventListener(LAB.TouchEvent.PRESS, onOutEvent);
 65 					break;
 66 			}
 67 		}
 68 	}
 69 		
 70 	// this could be useful as a public method as well
 71 	/**
 72 	 * @private
 73 	 */
 74 	function hitTest(event) {
 75 		//console.log(event);
 76 		var vec,
 77 			ray,
 78 			intersects;
 79 	
 80 		vec = new THREE.Vector3( (event.target.getTouchX() / window.innerWidth ) * 2 - 1, - (event.target.getTouchY() / window.innerHeight ) * 2 + 1, 0.5 );
 81 		projector.unprojectVector( vec, camera );
 82 		
 83 		ray = new THREE.Ray( camera.position, vec.subSelf( camera.position ).normalize() );
 84 		
 85 		intersects = ray.intersectObjects( scene.children );
 86 				
 87 		return intersects;	
 88 	}
 89 	
 90 	/**
 91 	 * @privte
 92 	 */
 93 	function handleEvent(type, evt) {
 94 		var i,
 95 			len,
 96 			intersects,
 97 			hitObject,
 98 			events,
 99 			params;
100 
101 		intersects = hitTest(evt);
102 		len = intersects.length;
103 		
104 		// if the hit test returned any objects
105 		if (len > 0) {
106 
107 			// iterate through array of intersected objects
108 			for (i = 0; i < len; i++) {
109 			
110 				hitObject = intersects[i].object;
111 				
112 				// this could be useful when dealing with dense scenes in determining which
113 				// is the closest object
114 				// may also be useful to implement event bubbling
115 				//console.log("distance = " + intersects[i].distance);
116 				
117 				// make sure the object is registered with the event handler
118 				if (hash[hitObject.id]) {
119 					// make sure the objects match (this is an extra precaution and may not
120 					// be necessary)
121 					if (hitObject === hash[hitObject.id].object) {
122 						// if this event type is registered for an object, fire the corresponding
123 						// event, passing a reference to the object					
124 						events = hash[hitObject.id].events;
125 						if (events != undefined && events.indexOf(type) > -1) {
126 							params = {object: hitObject, data: evt};
127 							self.dispatchEvent(new LAB.three.TouchEvent3D(type), params);
128 						}
129 					} else {
130 						console.log("no match");
131 					}
132 				}
133 				
134 			}
135 		}	
136 	}

139 	 * @private
140 	 */
141 	function onTouchTap(evt) {
142 		handleEvent(LAB.three.TouchEvent3D.TAP, evt);
143 	}

177 	function onOutEvent(evt) {

179 	}

181 	/**
182 	 * Register mouse or touch events with a 3D object (must inherit from THREE.Mesh).
183 	 * You can also use this method to register additional event types for a specific object.
184 	 *
185 	 * @param {Object} Object3D A 3D object to register events for.
186 	 * @param {String[]} events An array of event types (TouchEvent3D) to register.
187 	 */
188 	this.register = function(Object3D, events /* {String[]} */) {
189 		
190 		var len = events.length;
191 		// To Do: make sure the object is a THREE.Mesh. If not, throw error
192 		
193 		
194 		for (var j=0; j<len; j++) {
195 			// only start add event listeners as we need them
196 			addGestureEventListener(events[j]);
197 		}
198 		
199 		
200 		// To Do: only add listeners on this class for events that are registered
201 		// so if not 3D object listens for mouse move, then don't register mouse move?
202 		
203 		var registeredEvents,
204 			index;
205 		
206 		// this will only work if the object clicked on only has one unique mesh id
207 		if (!hash[Object3D.id]) {
208 			hash[Object3D.id] = {object: Object3D, events: events};
209 			//console.log(hash);
210 		} else {
211 			//if the object is already registered, append any new events
212 			registeredEvents = hash[Object3D.id].events;
213 			// iterate through events array
214 			for (var i=0; i<len; i++) {
215 				// if the event is not registered, add it
216 				if (registeredEvents.indexOf(events[i] == -1)) {
217 					registeredEvents.push(events[i]);
218 				}
219 			}
220 		}
221 	}
222 	
223 	
224 	/**
225 	 * Unregister mouse or touch events with a 3D object (must inherit from THREE.Mesh).
226 	 * Passing the object only with no events will unregister all events. Alternatively
227 	 * you can pass an arrayof one or more specific event types to unregister.
228 	 *
229 	 * @param {Object} Object3D A 3D object to unregister events for.
230 	 * @param {String[]} events An array of event types (TouchEvent3D) to unregister.
231 	 */	
232 	this.unregister = function(Object3D, events /* {String[]} */) {
233 	
234 		// To Do: make sure the object is a THREE.Mesh. If not, throw error 
235 	
236 		var events = events || null;
237 		var obj,
238 			index;
239 		
240 		// ensure the object is registered
241 		if (hash[Object3D.id]) {
242 			// if specific events are not unregistered, then unregister the object (delete key)
243 			if (events == null) {
244 				delete hash[Object3D.id];
245 			} else {
246 				// unregister individual events
247 				obj = hash[Object3D.id];
248 				//remove specified event type(s) from array
249 				for (var i=0, len=events.length; i<len; i++) {
250 					index = obj.events.indexOf(events[i]);
251 					if (index > -1) {
252 						obj.events.splice(index, 1);
253 					}
254 				}
255 			}
256 		}
257 		
258 	}
259 	
260 }
