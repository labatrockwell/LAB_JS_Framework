 15 LAB.tdl.Shader.prototype = {

 17 	setup : function ( vertexShaderID, fragmentShaderID ) {
 18       
 19 		var vertexShader = this.getShader( vertexShaderID );
 20 		var fragmentShader = this.getShader( fragmentShaderID );
 21       
 22 		this.program = gl.createProgram();
 23       gl.attachShader(this.program, vertexShader);
 24       gl.attachShader(this.program, fragmentShader);
 25       gl.linkProgram(this.program);
 26       
 27       if (!gl.getProgramParameter(this.program, gl.LINK_STATUS)) {
 28          alert("Could not initialise shaders");
 29       }
 30       
 31       gl.useProgram(this.program);
 32       
 33       //add typical attributes(if they exist). might be confusing down the road....
 34       this.addAttriubte("aColor");
 35       this.addAttriubte("aPosition");
 36       this.addAttriubte("aNormal");
 37       this.addAttriubte("aTexCoord");
 38       
 39       this.pMatrixUniform = this.addUniform( "projectionMatrix" );
 40       this.mvMatrixUniform = this.addUniform( "modelviewMatrix" );
 41       
 42       gl.useProgram(null);      
 43 	},
 44    
 45 	getShader : function ( id ) {
 46       var shaderScript = document.getElementById(id);
 47       if (!shaderScript) {
 48          return null;
 49       }
 50       
 51       var str = "";
 52       var k = shaderScript.firstChild;
 53       while (k) {
 54          if (k.nodeType == 3) {
 55             str += k.textContent;
 56          }
 57          k = k.nextSibling;
 58       }
 59       
 60       var shader;
 61       if (shaderScript.type == "x-shader/x-fragment") {
 62          shader = gl.createShader(gl.FRAGMENT_SHADER);
 63       } else if (shaderScript.type == "x-shader/x-vertex") {
 64          shader = gl.createShader(gl.VERTEX_SHADER);
 65       } else {
 66          return null;
 67       }
 68       
 69       gl.shaderSource(shader, str);
 70       gl.compileShader(shader);
 71       
 72       if (!gl.getShaderParameter(shader, gl.COMPILE_STATUS)) {
 73          alert(gl.getShaderInfoLog(shader));
 74          return null;
 75       }
 76       
 77       return shader;
 78 	},
 79

100    addAttriubte : function ( attrname ){
101       var attrloc = gl.getAttribLocation(this.program, attrname); 
102       if(attrloc !== -1){
103          this.attributes.push( attrloc );
104 
105       }else{
106          LAB.log( "no attribute named " + attrname +" in this shader" );
107       }
108       return attrloc;      
109    },

111    addAttribute : function ( attrname ){
112       var attrloc = gl.getAttribLocation(this.program, attrname); 
113       if(attrloc !== -1){
114          this.attributes.push( attrloc );
115 
116       }else{
117          LAB.log( "no attribute named " + attrname +" in this shader" );
118       }
119       return attrloc;      
120    },
121    
122    addUniform : function ( uniformname ){
123       var uniloc = gl.getUniformLocation( this.program, uniformname );
124       if(uniloc != -1){
125          uniloc.name = uniformname;
126          this.uniforms.push( uniloc );
127          return uniloc;
128       }else{
129          //no uniform of that name
130          return uniloc;//null;
131       }
132    },
