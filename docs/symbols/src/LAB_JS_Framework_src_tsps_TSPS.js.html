 31 TSPS.Dispatcher.prototype.parseMessage		= function(data)
 32 {
 33 	this.latestPeople = [];			
 34 
 35 	// parse message
 36 	var allMessages = data.split("\\");
 37 	for (var h=0; h<allMessages.length; h++){
 38 		var TSPSObj = new TSPS.person();
 39 
 40 		var dataArray = allMessages[h].split(";");
 41 		if (dataArray.length > 0){
 42 			TSPSObj.type = dataArray[0];
 43 			for (var i=1; i<dataArray.length; i++){
 44 				var dObj = dataArray[i].split("/");
 45 				if (dObj.length < 2){
 46 
 47 				} else if (dObj[1].indexOf(",") != -1){
 48 					var unpked = dObj[1].split(",");
 49 					TSPSObj[dObj[0]] = [];
 50 					for (var j=0; j<unpked.length; j++){
 51 						TSPSObj[dObj[0]][j] = {};
 52 						var obj = unpked[j].split(":");
 53 						for (var k=0; k<obj.length; k++){
 54 							var point = obj[k].split(">");
 55 							TSPSObj[dObj[0]][j][point[0]] = point[1];								
 56 						}
 57 					}
 58 				} else if (dObj[1].indexOf(":") != -1){
 59 					var unpked = dObj[1].split(":");
 60 					TSPSObj[dObj[0]] = {};
 61 
 62 					for (var j=0; j<unpked.length; j++){
 63 						var split = unpked[j].split(">");
 64 						TSPSObj[dObj[0]][split[0]] = split[1];
 65 					}						
 66 				} else {
 67 					TSPSObj[dObj[0]] = dObj[1];						
 68 				}
 69 			}
 70 
 71 			if (TSPSObj.type == "TSPS/personEntered/") this._onPersonEntered(TSPSObj);
 72 			else if (TSPSObj.type == "TSPS/personMoved/") this._onPersonUpdated(TSPSObj);
 73 			else if (TSPSObj.type == "TSPS/personUpdated/") this._onPersonMoved(TSPSObj);
 74 			else if (TSPSObj.type == "TSPS/personWillLeave/") this._onPersonLeft(TSPSObj);
 75 		}
 76 
 77 		this.latestPeople[this.latestPeople.length] = TSPSObj;
 78 		//console.log(TSPSObj);
 79 	}
 80 
 81 	return this.latestPeople;
 82 }
