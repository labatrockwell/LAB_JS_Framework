  5 /**
  6  * A touch gesture handler that wraps mouse events. To use: create an instance of this
  7  * class and register either window or a DOM object (or any custom object that implements
  8  * EventDispatcher). Then listen for touch events (see TouchEvent) on the registerd object.
  9  *
 10  * @namespace LAB
 11  * @constructor
 12  * @augments LAB.EventDispatcher
 13  */
 14 LAB.TouchGestureHandler = function() {
 15 	"use strict";
 16 
 17 	LAB.EventDispatcher.call(this, this);
 18 
 19 	// private constants
 20 	// tweak these to fine tune the gesture behavior
 21 	var FLICK_TIMEOUT = 250,
 22 		TAP_TIMEOUT = 80,
 23 		FLICK_MIN_MOVEMENT = 20,
 24 		DRAG_MIN_MOVEMENT = 5;
 25 		
 26 	// to do: add getters and setters for time and movement thresholds	
 27 		
 28 	// private variables
 29 	var _self = this,
 30 		_target = null,
 31 		_isDrag = false,
 32 		_isFlick = false,
 33 		_pressPoint,
 34 		_flickDirection,
 35 		_flickDistance,
 36 		_flickStartPoint,
 37 		_flickEndPoint,
 38 		_flickVelocity,
 39 		_flickStartTime,
 40 		_startTime,
 41 		_endTime,
 42 		_lastTime = -1,
 43 		_averageVelocity = [],
 44 		_velocity = [],
 45 		_dragDirection,
 46 		_dragDistance,
 47 		_touchX,
 48 		_touchY;
 49 	
 50 	/**
 51 	 * @private
 52 	 */
 53 	function mouseDownHandler(event) {
 54 	
 55 		// events to listen to
 56 		window.addEventListener("mouseup", mouseUpHandler);
 57 		_target.addEventListener("mousemove", mouseMoveHandler);
 58 		
 59 		// events to remove
 60 		_target.removeEventListener("mousedown", mouseDownHandler);
 61 		
 62 		startMove(event);
 63 	}
 64 	
 65 	/**
 66 	 * @private
 67 	 */
 68 	function startMove(event) {
 69 		_touchX = event.clientX;
 70 		_touchY = event.clientY;
 71 
 72 		dispatch(LAB.TouchEvent.PRESS);
 73 		
 74 		var date = new Date();
 75 		_startTime = date.getTime();
 76 						
 77 		_pressPoint = new LAB.geom.Point(event.clientX, event.clientY);
 78 	}
 79 	
 80 	/**
 81 	 * @private
 82 	 */
 83 	function mouseMoveHandler(event) {
 84 		
 85 		_touchX = event.clientX;
 86 		_touchY = event.clientY;
 87 		
 88 		var date = new Date();
 89 		
 90 		var currentTime = date.getTime();
 91 				
 92 		// save current point
 93 		var currentPoint = new LAB.geom.Point(_touchX, _touchY);
 94 		
 95 		// flick handling
 96 		var distance = calcDistance(currentPoint, _pressPoint);
 97 		var direction = calcDegree(currentPoint, _pressPoint);
 98 		
 99 		if (distance > FLICK_MIN_MOVEMENT) {
100 		
101 			var date;
102 		
103 			if (!_isFlick) {
104 				date = new Date();
105 				// get time and position the first time the flick movement threshold is met
106 				_flickStartTime = date.getTime();
107 				_flickStartPoint = new LAB.geom.Point(_touchX, _touchY);
108 			}
109 			
110 			_isFlick = true;
111 		}
112 				
113 		// drag handler
114 		// to do: only send drag events when an object is being dragged?
115 		if (distance > DRAG_MIN_MOVEMENT) {
116 			var velocity;
117 			// update the distance and direction each time the minimum drag movement
118 			// threshold is met
119 			_dragDistance = distance;
120 			_dragDirection = direction;
121 			
122 			var timeDiff = currentTime - _lastTime;
123 			
124 			velocity = calcVelocity(_dragDistance, timeDiff);
125 			if (velocity > -1) {
126 				_velocity.push(velocity);
127 			}
128 		
129 			_isDrag = true;
130 		}
131 		
132 		// to do: dispatch only on distance > 0?
133 		if (_isDrag) {
134 			// currently fired on each move event once the drag threshold has been met
135 			dispatch(LAB.TouchEvent.DRAG);
136 		}
137 		
138 		// update press point so we can have drag...drag...drag...flick
139 		_pressPoint = new LAB.geom.Point(currentPoint.x, currentPoint.y);
140 		_lastTime = currentTime;
141 	}
142 	
143 	/**
144 	 * This is important when using DOM objects so you know when you have
145 	 * dragged outside the object that the mouse down event was fired over
146 	 *
147 	 * @private
148 	 */
149 	function mouseOutHandler(event) {		
150 		// ignore if out is not this object
151 		if (event.target != _target) {
152 			return;
153 		}
154 		
155 		_target.removeEventListener("mouseout", mouseOutHandler);
156 		
157 		dispatch(LAB.TouchEvent.OUT);
158 	}	
159 	
160 	/**
161 	 * @private
162 	 */
163 	function mouseUpHandler(event) {
164 	
165 		_touchX = event.clientX;
166 		_touchY = event.clientY;
167 		
168 		_target.addEventListener("mousedown", mouseDownHandler);
169 		
170 		_target.removeEventListener("mousemove", mouseMoveHandler);
171 		window.removeEventListener("mouseup", mouseUpHandler);
172 		
173 		var date = new Date();
174 		_endTime = date.getTime();
175 		
176 		var timeChange = _endTime - _startTime;
177 		var flickTimeChange = _endTime - _flickStartTime;
178 		
179 		if (timeChange <= TAP_TIMEOUT && !_isDrag && !_isFlick) {
180 			dispatch(LAB.TouchEvent.TAP);
181 		}
182 		
183 		var velocitySum = 0;
184 		for (var i = 0, len = _velocity.length; i< len; i++) {
185 			velocitySum = velocitySum + _velocity[i];
186 		}
187 		_averageVelocity = velocitySum / _velocity.length;
188 		
189 		// must meet minimum time requirement for flick
190 		if (flickTimeChange <= FLICK_TIMEOUT && _isFlick) {
191 			
192 			_flickEndPoint = new LAB.geom.Point(_touchX, _touchY);
193 			_flickDirection = calcDegree(_flickEndPoint, _flickStartPoint);
194 			_flickDistance = calcDistance(_flickEndPoint, _flickStartPoint);
195 			_flickVelocity = calcVelocity(_flickDistance, timeChange);
196 		
197 			dispatch(LAB.TouchEvent.FLICK);
198 		}
199 		
200 		// always dispatch release whether or not release happened over target object
201 		dispatch(LAB.TouchEvent.RELEASE);
202 		// or only fire release event if release occured over target?
203 		// hit test for top most element		
204 		//if (_target == window || _target === document.elementFromPoint(_touchX, _touchY)) {								
205 			//dispatch(LAB.TouchEvent.RELEASE);
206 		//}
207 		
208 		resetForNext();	
209 	}
210 	
211 	/**
212 	 * @private
213 	 */
214 	function calcDistance(pointA, pointB) {
215 		var dist = Math.sqrt(Math.pow((pointB.x - pointA.x), 2) + Math.pow((pointB.y - pointA.y), 2));
216 		
217 		return dist;
218 	}
219 	
220 	/**
221 	 * @private
222 	 */
223 	function calcVelocity(dist, timeChange) {
224 		if (dist > 0 && timeChange > 0) {
225 			return dist / timeChange
226 		} else {
227 			return -1;
228 		}
229 	}
230 	
231 	/**
232 	 * @private
233 	 */
234 	function dispatch(type) {
235 		_self.dispatchEvent(new LAB.TouchEvent(type));
236 		
237 		// any way or reason to get something like this to work?
238 		// seems tricky for window and DOM events, need to use
239 		// customEvent or something like that, but it does not
240 		// seem to support additional parameters
241 		//_target.dispatchEvent(new LAB.TouchEvent(type, _self));
242 	}
243 	
244 	/**
245 	 * @private
246 	 */
247 	function resetForNext() {		
248 		_isDrag = false;
249 		_isFlick = false;
250 		
251 		// clear the array
252 		_averageVelocity = [];		
253 	}
254 	
255 	/**
256 	 * @private
257 	 */
258 	function calcDegree(startPoint, endPoint) {
259 		var diffX = startPoint.x - endPoint.x;
260 		var diffY = endPoint.y - startPoint.y;
261 		
262 		return tangentAngle(diffX, diffY);
263 	}
264 	
265 	/**
266 	 * @private
267 	 */
268 	function tangentAngle(xVal, yVal) {
269 		var radians = Math.atan2(yVal, xVal);
270 		
271 		// handle going all the way around
272 		if (radians < 0) radians += 6.283185;
273 		
274 		// convert radians to degrees
275 		return radians * 57.29578;
276 	}
277 	
278 		
279 	/**
280 	 * Register an object (DOM, window, or any object that implements EventDispatcher) with
281 	 * the touch gesture handler.
282 	 *
283 	 * @param {Object} target The object to register with the gesture handler
284 	 */
285 	this.register = function(target) {
286 		_target = target;
287 		_target.addEventListener("mousedown", mouseDownHandler);
288 	}
289 	
290 	/**
291 	 * Unregister an object with the touch gesture handler
292 	 *
293 	 * @param {Object} target The object to unregister
294 	 */
295 	this.unregister = function(target) {
296 		if (_target !== null) {
297 			_target.removeEventListener("mousedown", mouseDownHandler);
298 			_target.removeEventListener("mouseout", mouseOutHandler);
299 			_target.removeEventListener("mousemove", mouseMoveHandler);
300 			window.removeEventListener("mouseup", mouseUpHandler);
301 		}
302 	}
303 	
304 	/**
305 	 * @return {Number} The direction (in degrees 0 - 360) between the start and stop point 
306 	 * of the last drag event.
307 	 */
308 	this.getDragDirection = function() {
309 		return _dragDirection;
310 	}
311 	
312 	/**
313 	 * @return {Number} The distance between the start and stop point of the last drag event.
314 	 */
315 	this.getDragDistance = function() {
316 		return _dragDistance;
317 	}
318 	
319 	/**
320 	 * @return {Number} The direction (in degrees 0 - 360) between the start and stop point 
321 	 * of the flick
322 	 */
323 	this.getFlickDirection = function() {
324 		return _flickDirection;
325 	}
326 	
327 	/**
328 	 * @return {Number} The distance between the start and end point of the flick.
329 	 */
330 	this.getFlickDistance = function() {
331 		return _flickDistance;
332 	}	
333 	
334 	/**
335 	 * @return {Number} The x coordinate of the touch event.
336 	 */
337 	this.getTouchX = function() {
338 		return _touchX;
339 	}
340 	
341 	/**
342 	 * @return {Number} The y coordinate of the touch event.
343 	 */
344 	this.getTouchY = function() {
345 		return _touchY;
346 	}
347 	
348 	/**
349 	 * @return {Number} The average velocity (in pixels/milliseconds) of the drag 
350 	 * from press to release.
351 	 */
352 	this.getAverageVelocity = function() {
353 		return _averageVelocity;
354 	}
355 	
356 	/**
357 	 * @return {Number} The velocity (in pixels/milliseconds) of the flick.
358 	 */
359 	this.getFlickVelocity = function() {
360 		return _flickVelocity;
361 	}	
362 	
363 }

369 /**

371  * @constructor
372  * @augments LAB.Event
373  * @param {String} type The event type

375 LAB.TouchEvent = function(type) {
376 	"use strict";
377 	
378 	LAB.Event.call(this, type);
379 }
