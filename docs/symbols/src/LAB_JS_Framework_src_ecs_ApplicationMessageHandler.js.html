 80 			// throw away auth for now

 90 	        var type = xmlDoc.getElementsByTagName("type")[0];
 91 	        var messType = type.childNodes[0].nodeValue;
 92 	        if (bVerbose) console.log("type: " + messType);
 93 
 94 
 95 	        // Call the appropriate handlers based on the type of this message

106 	            var configs = xmlDoc.getElementsByTagName("configs")[0];

111 	                var a_key 		= null;
112 	                var a_value 	= null;
113 	                var children = curList[i].childNodes;
114 
115 	                // Iterate through all child tags (i.e., <key>  and  <value>  tags).
116 	                for (var j=0; j < children.length; j++) {
117 	                    var curChild = children[j];
118 
119 	                    if ("key" == curChild.nodeName) {
120 	                        a_key = curChild.childNodes[0].nodeValue;                        
121 	                    } else if ("value" == curChild.nodeName) {

123 	                    }

134 	            var configs = xmlDoc.getElementsByTagName("configs")[0];

142 				var cont	= "";
143 				var app		= "";

145 				for (var i=0; i<dest.childNodes.length; i++){
146 					if (dest.childNodes[i].childNodes.length > 0){
147 						if (dest.childNodes[i].nodeName == "type"){
148 							type	= dest.childNodes[i].childNodes[0].nodeValue;
149 						} else if (dest.childNodes[i].nodeName == "room"){

151 						} else if (dest.childNodes[i].nodeName == "controller"){

369 		// private vars
370         var _self       		=   this;
371 		var _port;		// this comes from the controller

373 		var _outSocket;

381 	    this.name               =   'ECS_Connection';
382 
383 		this.connect = function(port, server){
384 			_port 	= port;
385 			if (!server) server = _server;
386 			_server = server;
387 
388 			//*
389 			var serverStr = 'ws://' + _server + ':' + _port;
390 
391 			if (bVerbose) console.log('Attempting connection to ' + serverStr);
392 			//ws://localhost:60600
393 
394 			_outSocket = new WebSocket(serverStr);
395 			_outSocket.onopen = function () {
396 				if (bVerbose) console.log('port open!');
397 				_self.dispatchEvent("open", null);
398 				bConnected	= true;
399 			};
400 
401 			_outSocket.onmessage = function(msg){
402 				if (bVerbose) console.log('got message:', msg);
403 				_self.dispatchEvent("message", msg.data);
404 			};
405 
406 			_outSocket.onclose = function(){
407 				if (bVerbose) console.log('socket connection closed');
408 				_self.dispatchEvent("close", null);
409 				bConnected	= false;
410 			}
411 
412 		};
413 
414 		this.send	= function (data)
415 		{
416 			//this could be smarter, but for now directly send
417 			if (bConnected){				
418 				var bodyLen = data.length;
419 				var headerLen = 0;
420 				var tempLength = bodyLen;
421 				while (tempLength > 0) {
422 				  tempLength = Math.floor( tempLength / 10);
423 				  headerLen++;
424 				}
425 
426 				var _messageLength = 1 + headerLen + bodyLen;
427 				var _message = headerLen + "" + bodyLen + data;
428 
429 				_outSocket.send(_message);
430 				if (bVerbose) console.log("sending "+_message);
431 			}
432 		}
