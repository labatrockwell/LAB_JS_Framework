 58    var copy = new Float32Array(16);
 59    tdl.fast.matrix4.copy(copy, LAB.tdl.modelviewMatrix);
 60    LAB.tdl.mvMatrixStack.push(copy);

146 LAB.tdl.setSceneTile	= function( scenePos, sceneScl, canvasPos, canvasDim ){
147 	gl.enable(gl.SCISSOR_TEST);
148 	gl.viewport(-scenePos[0]*sceneScl + canvasPos[0],// -position + screenposition(centers the view)
149               -scenePos[1]*sceneScl + canvasPos[1],
150               total_width*sceneScl,
151               total_height*sceneScl);
152 	gl.scissor(canvasPos[0], canvasPos[1], canvasDim[0], canvasDim[1]);
153 	gl.disable(gl.SCISSOR_TEST);
154 }

178 LAB.tdl.setup	= function(){
179    //math and matrix
180    math = tdl.math;
181    fast = tdl.fast;
182    
183    //setup canvas and context
184    LAB.tdl.canvas = document.getElementById("canvas");
185    LAB.tdl.g_fpsTimer = new tdl.fps.FPSTimer();
186    
187    gl = tdl.webgl.setupWebGL(LAB.tdl.canvas);
188    if (!gl) {
189       return false;
190    }
191    if (LAB.tdl.g_debug) {
192       gl = tdl.webgl.makeDebugContext(gl, undefined, LogglCall);
193    }
194    
195    LAB.tdl.setMatrices();
196 }

198 LAB.tdl.update	= function(){
199    
200    var now = (new Date()).getTime();
201    if(LAB.tdl.lastTime == 0.0) {
202       LAB.tdl.deltaTime = 0.0;
203    } else {
204       LAB.tdl.deltaTime = now - LAB.tdl.lastTime;
205    }
206    LAB.tdl.lastTime = now;
207    LAB.tdl.labTimeDelta = deltaTime;
208    
209    LAB.tdl.g_fpsTimer.update(deltaTime * .001);//<--- to milliseconds?
210    if (fpsElem != null){
211 		fpsElem.innerHTML = LAB.tdl.g_fpsTimer.averageFPS;		
212 	} 
213    
214    LAB.tdl.labFPS = LAB.tdl.g_fpsTimer.instantaneousFPS;
215    LAB.tdl.elapsedTime += LAB.tdl.deltaTime;
216 }

218 LAB.tdl.setMatrices	= function(){
219    //taken from openframeworks
220    LAB.tdl.eyePosition[0] = LAB.tdl.getCanvasWidth()/2;
221    LAB.tdl.eyePosition[1] = LAB.tdl.getCanvasHeight()/2;
222    LAB.tdl.eyePosition[2] = LAB.tdl.eyePosition[1] / Math.tan( Math.PI * LAB.tdl.fov / 360 );
223    LAB.tdl.aspect = LAB.tdl.getCanvasWidth() / LAB.tdl.getCanvasHeight();
224    LAB.tdl.target[0] = LAB.tdl.eyePosition[0];
225    LAB.tdl.target[1] = LAB.tdl.eyePosition[1];
226    LAB.tdl.target[2] = 0;
227    
228    LAB.tdl.nearClip = LAB.tdl.eyePosition[2] / 10;
229    LAB.tdl.farClip = LAB.tdl.eyePosition[2] * 10;
230    
231    fast.identity4( LAB.tdl.projectionMatrix );
232    fast.matrix4.perspective( LAB.tdl.projectionMatrix, math.degToRad(LAB.tdl.fov), LAB.tdl.aspect, LAB.tdl.nearClip, LAB.tdl.farClip);
233    
234    fast.identity4(LAB.tdl.modelviewMatrix);
235    fast.matrix4.lookAt(LAB.tdl.modelviewMatrix, LAB.tdl.eyePosition, LAB.tdl.target, LAB.tdl.up);
236    
237    
238    //move to upper left corner and scale. mimics openframeworks setup
239    LAB.tdl.scale(-1,1,1);
240    LAB.tdl.translate(-LAB.tdl.getCanvasWidth(), 0, 0);
241 }

282 LAB.tdl.ValidateNoneOfTheArgsAreUndefined	= function(functionName, args) {
283    for (var ii = 0; ii < args.length; ++ii) {
284       if (args[ii] === undefined) {
285          tdl.error("undefined passed to gl." + functionName + "(" +
286          tdl.webgl.glFunctionArgsToString(functionName, args) + ")");
287       }
288    }
289 }

297 LAB.tdl.LogglCall	= function(functionName, args) {
298    if (LAB.tdl.g_logglCalls) {
299       ValidateNoneOfTheArgsAreUndefined(functionName, args)
300       tdl.log("gl." + functionName + "(" +
301       tdl.webgl.glFunctionArgsToString(functionName, args) + ")");
302    }
303 }

317 LAB.tdl.labRandom	= function( min, max ){
318    return tdl.math.pseudoRandom()*(max-min) + min;
319 }
320 
321 LAB.tdl.labMap	= function(value, oldMin, oldMax, min, max){    
322    return min + ((value-oldMin)/(oldMax-oldMin)) * (max-min);
323 }
