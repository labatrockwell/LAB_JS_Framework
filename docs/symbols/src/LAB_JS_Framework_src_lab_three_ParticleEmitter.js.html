 27 LAB.three.ParticleEmitter = function ( parameters ) {
 28    parameters = parameters || {};
 29    
 30    //create particle data. this'll be crushed into arrays in the renderer
 31    this.maxParticleCount = parameters.maxParticleCount || 5000;
 32    this.particles = [];
 33    this.scene = parameters.scene || new THREE.Scene();
 34    this.renderer = parameters.renderer || new THREE.WebGLRenderer();
 35    this.camera = parameters.camera || new THREE.PerspectiveCamera(60,window.innerWidth/window.innerHeight);
 36    this.currentTime = 0;
 37    
 38    for(var i=0; i<this.maxParticleCount; i++){
 39       if(parameters.createParticle){
 40          //I used this to make particles with additional variables and methds
 41          this.particles[i] = parameters.createParticle();
 42       }
 43       else this.particles[i] = new labParticle();
 44    }
 45    
 46    //geometry and rendering
 47    this.attributes = {
 48    pColor: {type: "v3", value: []},
 49    radius: {type: "f", value: []},
 50    birth: {type: "f", value: []},
 51    lifespan: {type: "f", value: []},
 52    };
 53    this.geometry = new THREE.Geometry();
 54    
 55    for(var i=0; i<this.particles.length;i++){
 56       this.geometry.vertices[i] = new THREE.Vertex( this.particles[i].pos );
 57       this.attributes.pColor.value[i] = new THREE.Vector3(1, 0, 0);//this.particles[i].col;
 58       this.particles[i].col = this.attributes.pColor.value[i];
 59       this.attributes.radius.value[i] = this.particles[i].radius;
 60       this.attributes.birth.value[i] = this.particles[i].birth;
 61       this.attributes.lifespan.value[i] = this.particles[i].lifespan;
 62    }
 63    this.geometry.dynamic = true;
 64    this.geometry.__dirtyVertices = true;
 65    
 66    //TODO: we need to write the defualt shader with text
 67    var shaderUniforms = {
 68       tex:   { type: "t", value: 0, texture: THREE.ImageUtils.loadTexture( "textures/sphereNormal.png" )},
 69       currentTime:   { type: "f", value: 0},
 70       pScl: {type: "f", value: 100},
 71    };
 72    
 73    this.shader = new LAB.three.Shader({name: "shaders/basicParticle",
 74                                       uniforms: shaderUniforms,
 75                                       attributes: this.attributes,
 76                                       });
 77    
 78    this.particleSystem = new THREE.ParticleSystem( this.geometry, this.shader );
 79       
 80    this.scene.add( this.particleSystem );
 81    this.renderer.render( this.scene, this.camera );//this is kinda sloppy, but it's any easy way to create the webgl buffers
 82    this.particleSystem.geometry.__webglParticleCount = 0;// set the particle count to 0
 83    
 84 };

137 LAB.three.ParticleEmitter.prototype.addParticle = function( pos, vel, col, radius, currentTime, lifespan) {
138    if(this.particleSystem.geometry.__webglParticleCount < this.maxParticleCount - 1){
139       var i = this.particleSystem.geometry.__webglParticleCount;
140       this.particles[i].pos.copy( pos || {x:0, y:0, z:0} );
141       this.particles[i].vel.copy( vel || {x:0, y:0, z:0} );
142       this.particles[i].col.copy( col || {x:1, y:1, z:1} );
143       
144       this.particles[i].radius = this.attributes.radius.value[i] = radius || 10;
145       this.particles[i].birth = this.attributes.birth.value[i] = currentTime || LAB.getElapsedTimeSeconds();
146       this.particles[i].lifespan = this.attributes.lifespan.value[i] = lifespan || 10;
147 
148       
149       this.particleSystem.geometry.__webglParticleCount++;
150       this.geometry.__dirtyVertices = true;
151       this.attributes.pColor.needsUpdate = true;
152       this.attributes.radius.needsUpdate = true;
153       this.attributes.birth.needsUpdate = true;
154       this.attributes.lifespan.needsUpdate = true;
155       
156       
157       return this.particles[i];
158    }
159    else return null;
160 };

162 LAB.three.ParticleEmitter.prototype.removeParticle = function( index ){
163    
164    if(this.geometry.__webglParticleCount > 0){
165       
166       this.geometry.__webglParticleCount -= 1;
167 
168       this.particles[index].copy( this.particles[ this.geometry.__webglParticleCount ] );
169       
170       this.attributes.radius.value[index] = this.particles[index].radius;
171       this.attributes.lifespan.value[index] = this.particles[index].lifespan;
172       this.attributes.birth.value[index] = this.particles[index].birth;
173       
174       this.geometry.__dirtyVertices = true;
175       this.attributes.pColor.needsUpdate = true;
176       this.attributes.radius.needsUpdate = true;
177       this.attributes.birth.needsUpdate = true;
178       this.attributes.lifespan.needsUpdate = true;
179       
180    }
181 };
