 19 /**
 20  * Copied from Flanagan, David. Javascript: The Definitive Guide.
 21  * Use this function to inherit objects.
 22  *
 23  * @function
 24  * @example
 25  * Child.prototype = LAB.inherit(Parent.prototype);
 26  * Child.prototype.constructor = Child;
 27  *
 28  * @param {Object} p The Parent object to inherit. The object can either be the prototype of the Parent
 29  * object (Parent.prototype) or a new instance of the parent (new Parent())
 30  * @return {Object} A copy of the prototype of the Parent object.
 31  */
 32 LAB.inherit = function(p) {
 33 	if (p == null) return; // p must be a non-null object
 34 	if (Object.create) { // If Object.create() is defined...
 35 		return Object.create(p); // then just use it
 36 	}
 37 	var t = typeof p; // otherwise do some more type checking
 38 	if (t !== "object" && t !== "function") throw TypeError();
 39 	function f() {}; // define a dummy constructor function
 40 	f.prototype = p; // Set its prototype property to p
 41 	return new f(); // use f() to create an 'heir' of p.
 42 };

 64 /**
 65  * Use this function to safely create a new namespace
 66  * if a namespace already exists, it won't be recreated.
 67  *
 68  * @function
 69  * @param {String} nsString The namespace as a string.
 70  * @return {Object} The namespace object.
 71  */
 72 LAB.namespace = function (nsString) {
 73 	var parts = nsString.split('.'),
 74 		parent = LAB,
 75 		i;
 76 		
 77 	// strip redundant leading global
 78 	if (parts[0] === "LAB") {
 79 		parts = parts.slice(1);
 80 	}
 81 	
 82 	for (i=0; i<parts.length; i +=1) {
 83 		// create a property if it doesn't exist
 84 		if (typeof parent[parts[i]] === "undefined") {
 85 			parent[parts[i]] = {};
 86 		}
 87 		parent = parent[parts[i]];
 88 	}
 89 	return parent;
 90 };

 96 LAB.writeScriptTag_ = function(src) {
 97 	var doc = LAB.global.document;
 98 	doc.write(
 99 		'<script type="text/javascript" src="' + src + '"></' + 'script>');
100 	included[src] = true;
101 	return true;
102 };

113 LAB.getScriptPath = function(filename) {
114 	var scripts = document.getElementsByTagName('script');
115 
116     for (var i = scripts.length - 1; i >= 0; --i) {
117 		var src = scripts[i].src;
118 		var l = src.length;
119 		var length = filename.length;
120 		if (src.substr(l - length) == filename) {
121         	// set a global propery here
122         	return src.substr(0, l - length);
123 		}
124     }
125 	return "";
126 };
