  9 LAB.three.Shader = function ( parameters ) {
 10   THREE.ShaderMaterial.call( this );
 11    
 12    parameters = parameters || {};
 13    if(parameters.name){
 14       $.ajax({ async: false, type: 'GET', url: parameters.name.concat( ".vert"),
 15              success: function(data) { parameters.vertexShader = data.slice( 0, data.length ); } });
 16       
 17       $.ajax({ async: false, type: 'GET', url: parameters.name.concat( ".frag"),
 18              success: function(data) { parameters.fragmentShader = data.slice( 0, data.length ); }});
 19       
 20       THREE.ShaderMaterial.call( this, parameters );
 21    }   
 22    
 23 };
