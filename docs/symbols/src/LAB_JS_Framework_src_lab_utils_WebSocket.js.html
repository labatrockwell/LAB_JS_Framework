  6 /** 
  7 	Creates a new Websocket with host (defaults to "ws://localhost:8888")
  8 	@constructor
  9 	@augments LAB.EventDispatcher
 10 */
 11 
 12 LAB.utils.WebSocket = function( _host, protocol ) {
 13 	this.host 		= _host || "ws://localhost:8888";
 14 	if (protocol) this.protocol	= protocol;
 15 	this.socket;
 16 }
 17 
 18 /**
 19 @function
 20 */
 21 LAB.utils.WebSocket.prototype.connect = function() {
 22 	try {
 23 		if (this.protocol) this.socket = new WebSocket( this.host, this.protocol );
 24 		else this.socket = new WebSocket( this.host);
 25 		this.socket._parent = this;
 26 		this.socket.onmessage = this._onMessageReceived;
 27 		this.socket.onopen = this._onConnectionOpened;
 28 		this.socket.onclose = this._onConnectionClosed;
 29 	} catch( ex ) {
 30 		console.log( ex );
 31 	}
 32 }
 33 
 34 /**
 35 @function
 36 */
 37 LAB.utils.WebSocket.prototype.sendData = function( data ) {
 38 	console.log( "sending: " + data );
 39 	this.socket.send( data );
 40 }
 41 
 42 LAB.utils.WebSocket.prototype._onConnectionOpened = function() { this._parent.onConnectionOpened(); }
 43 
 44 /**
 45 Override in your main app, e.g.:
 46 var socket = new LAB.utils.WebSocket;
 47 socket.onConnectionOpened = this.onConnect;
 48 @function
 49 */
 50 LAB.utils.WebSocket.prototype.onConnectionOpened = function() {};
 51 
 52 LAB.utils.WebSocket.prototype._onConnectionClosed = function() { this._parent.onConnectionClosed(); }
 53 
 54 /**
 55 Override in your main app
 56 @function
 57 */
 58 LAB.utils.WebSocket.prototype.onConnectionClosed = function() {};

 61 LAB.utils.WebSocket.prototype._onMessageReceived = function( evt ) {
 62 	var data = evt.data;
 63 	try {
 64 		var data =  jQuery.parseJSON( evt.data );	
 65 	} catch(e) {
 66 		// not valid JSON? Other reasons for error?
 67 	}
 68 	this._parent.onMessageReceived( data );
 69 }
