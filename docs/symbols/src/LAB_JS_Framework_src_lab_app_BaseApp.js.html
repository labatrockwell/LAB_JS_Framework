  8 /** 
  9 	Creates a new LAB base app. Extend this in your main app
 10 	or with a new LAB app class
 11 	@constructor
 12 	@augments LAB.EventDispatcher
 13 */
 14 
 15 LAB.app.BaseApp = function()
 16 {
 17 	LAB.EventDispatcher.call( this, this );
 18 	
 19 	/**
 20 		Mouse object: {x:Number, y:Number, bDown:Boolean}
 21 		@type Object
 22 	*/
 23 	this.mouse	= {x:0, y:0, bDown:false};
 24 	this.mouseEventsRegistered 	= false;
 25 	this.keyEventsRegistered 	= false;
 26 	this.windowEventsRegistered = false;
 27 	
 28 	this.startTime 		= Date.now();
 29 	this.elapsedTime 	= 0;
 30 }

 47 };

 96 /**
 97 	starts off animation + sets up preupdate-update-predraw-draw-postdraw chain
 98 	@function
 99 	@private
100 */
101 LAB.app.BaseApp.prototype.animate	= function(time){
102 	// update time
103 	if (time === undefined){
104 	} else {
105 		this.elapsedTime = time - this.startTime;
106 	}
107 	
108 	requestAnimationFrame(this.animate.bind(this), this);
109 			
110 	this.preupdate();
111    	this.update();
112 	this.predraw();
113 	this.draw();
114 	this.postdraw();
115 }

138 	/**
139 		start listening to mouse events
140 		@function
141 		@public
142 	*/
143 	LAB.app.BaseApp.prototype.registerMouseEvents = function(){
144 		if (this.mouseEventsRegistered) return;
145 		window.addEventListener("mousemove", this._onMouseMoved.bind(this));
146 		window.addEventListener("mousedown", this._onMousePressed.bind(this));
147 		window.addEventListener("mouseup", this._onMouseReleased.bind(this));
148 		this.mouseEventsRegistered = true;
149 	}
150 	/**
151 		stop listening to mouse events
152 		@function
153 		@public
154 	*/
155 	LAB.app.BaseApp.prototype.unregisterMouseEvents = function()
156 	{
157 		if (!this.mouseEventsRegistered) return;
158 		window.removeEventListener("mousemove", this._onMouseMoved.bind(this));
159 		window.removeEventListener("mousedown", this._onMousePressed.bind(this));
160 		window.removeEventListener("mouseup", this._onMouseReleased.bind(this));
161 		this.mouseEventsRegistered = false;
162 	}

201 	/**
202 		called directly by Window. override in your app to directly
203 		catch the "mousemove" event from Window.
204 		@function
205 		@private
206 		@param {Event} event DOM mouse event
207 	*/
208 
209 	LAB.app.BaseApp.prototype._onMouseMoved		= function( event )
210 	{	 
211 		// if the mouse is down, call dragged instead of moved
212 		if (this.mouse.bDown){
213 			this._onMouseDragged(event);
214 			return;
215 		}
216 		this.mouse.x = event.clientX; 
217 		this.mouse.y = event.clientY;
218 		this.dispatchEvent("onMouseMoved", this.mouse);
219 		this.onMouseMoved(this.mouse.x, this.mouse.y);
220 	}
221 
222 	/**
223 		called directly by Window. override in your app to directly
224 		catch the "mousedown" event from Window.
225 		@function
226 		@private
227 		@param {Event} event DOM mouse event
228 	*/
229 
230 	LAB.app.BaseApp.prototype._onMousePressed	= function( event )
231 	{
232 		this.mouse.x = event.clientX; 
233 		this.mouse.y = event.clientY;
234 		this.mouse.bDown = true;
235 		this.dispatchEvent("onMousePressed", this.mouse);
236 		this.onMousePressed(this.mouse.x, this.mouse.y);
237 	}
238 
239 	/**
240 		called directly by Window. override in your app to directly
241 		catch the "mouseup" event from Window.
242 		@function
243 		@private
244 		@param {Event} event DOM mouse event
245 	*/
246 
247 	LAB.app.BaseApp.prototype._onMouseReleased	= function( event )
248 	{
249 		this.mouse.x = event.clientX; 
250 		this.mouse.y = event.clientY;
251 		this.mouse.bDown = false;
252 		this.dispatchEvent("onMouseReleased", this.mouse);
253 		this.onMouseReleased(this.mouse.x, this.mouse.y);
254 	}
255 	
256 	/**
257 		helper function to call onMouseDragged instead of onMouseMoved.
258 		overriding this won't help anybody.
259 		@function
260 		@private
261 		@param {Event} event DOM mouse event
262 	*/ 
263 
264 	LAB.app.BaseApp.prototype._onMouseDragged	= function( event )
265 	{
266 		this.mouse.x = event.clientX; 
267 		this.mouse.y = event.clientY;
268 		this.dispatchEvent("onMouseDragged", this.mouse);
269 		this.onMouseDragged(this.mouse.x, this.mouse.y);
270 	}
