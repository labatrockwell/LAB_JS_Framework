LAB.three.Object = function( _renderer, _scene ) {
   
   this.renderer = _renderer;
   this.scene = _scene || new THREE.Scene();;
   this.autoClear = true;
   this.labObj = new THREE.Object3D();
   
   //this isn't great. temporary stop gap
   this.posStack = [];
   this.rotStack = [];
   this.sclStack = [];
   
   this.scene.add( this.labObj );
};
