  9 /** 
 10 	Creates a new LAB base app. Extend this in your main app
 11 	or with a new LAB app class
 12 	@constructor
 13 	@augments LAB.EventDispatcher
 14 */
 15 
 16 LAB.app.BaseApp = function()
 17 {
 18 	/**
 19 		Globally-accessible app scope
 20 		@type LAB.app.BaseApp
 21 	*/	
 22 	this 	= this;
 23 	LAB.EventDispatcher.call( this );
 24 	
 25 	/**
 26 		Mouse object: {x:Number, y:Number, bDown:Boolean}
 27 		@type Object
 28 	*/
 29 	this.mouse	= {x:0, y:0, bDown:false};
 30 	
 31 	this.startTime 		= Date.now();
 32 	this.elapsedTime 	= 0;
 33 }
 34

 36 LAB.app.BaseApp.prototype.constructor 	= LAB.app.BaseApp;
 37 LAB.app.BaseApp.prototype.supr 			= LAB.EventDispatcher.prototype;

 45 	console.log("base app set up");
 46 	this.setup();
 47 	this.animate();
 48 };

 97 /**
 98 	starts off animation + sets up preupdate-update-predraw-draw-postdraw chain
 99 	@function
100 	@private
101 */
102 LAB.app.BaseApp.prototype.animate	= function(time){
103 	// update time
104 	if (time === undefined){
105 	} else {
106 		this.elapsedTime = time - this.startTime;
107 	}
108 	
109 	requestAnimationFrame(this.animate, this);
110 			
111 	this.preupdate();
112    	this.update();
113 	this.predraw();
114 	this.draw();
115 	this.postdraw();
116 }

172 	/**
173 		called directly by Window. override in your app to directly
174 		catch the "mousemove" event from Window.
175 		@function
176 		@private
177 		@param {Event} event DOM mouse event
178 	*/
179 
180 	LAB.app.BaseApp.prototype._onMouseMoved		= function( event )
181 	{	 
182 		// if the mouse is down, call dragged instead of moved
183 		if (this.mouse.bDown){
184 			this.onMouseDragged(event);
185 			return;
186 		}
187 		this.mouse.x = event.clientX; 
188 		this.mouse.y = event.clientY;
189 		this.dispatchEvent("onMouseMoved", this.mouse);
190 		this.onMouseMoved(this.mouse.x, this.mouse.y);
191 	}
192 
193 	/**
194 		called directly by Window. override in your app to directly
195 		catch the "mousedown" event from Window.
196 		@function
197 		@private
198 		@param {Event} event DOM mouse event
199 	*/
200 
201 	LAB.app.BaseApp.prototype._onMousePressed	= function( event )
202 	{
203 		this.mouse.x = event.clientX; 
204 		this.mouse.y = event.clientY;
205 		this.mouse.bDown = true;
206 		this.dispatchEvent("onMousePressed", this.mouse);
207 		this.onMousePressed(this.mouse.x, this.mouse.y);
208 	}
209 
210 	/**
211 		called directly by Window. override in your app to directly
212 		catch the "mouseup" event from Window.
213 		@function
214 		@private
215 		@param {Event} event DOM mouse event
216 	*/
217 
218 	LAB.app.BaseApp.prototype._onMouseReleased	= function( event )
219 	{
220 		this.mouse.x = event.clientX; 
221 		this.mouse.y = event.clientY;
222 		this.mouse.bDown = false;
223 		this.dispatchEvent("onMouseReleased", this.mouse);
224 		this.onMouseReleased(this.mouse.x, this.mouse.y);
225 	}
226 	
227 	/**
228 		helper function to call onMouseDragged instead of onMouseMoved.
229 		overriding this won't help anybody.
230 		@function
231 		@private
232 		@param {Event} event DOM mouse event
233 	*/
234 
235 	LAB.app.BaseApp.prototype._onMouseDragged	= function( event )
236 	{
237 		this.mouse.x = event.clientX; 
238 		this.mouse.y = event.clientY;
239 		this.dispatchEvent("onMouseDragged", this.mouse);
240 		this.onMouseDragged(this.mouse.x, this.mouse.y);
241 	}
