1 /** @namespace LAB */
  2 LAB.EventDispatcher = function()
  3 {
  4 	this.events = [];
  5 }
  6 
  7 LAB.EventDispatcher.prototype = {
  8 	constructor:LAB.EventDispatcher,
  9 	addEventListener: function(event, callback){
 10 		this.events[event] = this.events[event] || []; // check for undefined arg
 11 		if (this.events[event]){
 12 			this.events[event].push(callback);
 13 		}
 14 	},
 15 	removeEventListener: function(event,callback){
 16 		if ( this.events[event] ) {
 17 			var listeners = this.events[event];
 18 			for ( var i = listeners.length-1; i>=0; --i ){
 19 				if ( listeners[i] === callback ) {
 20 					listeners.splice( i, 1 );
 21 					return true;
 22 				}
 23 			}
 24 		}
 25 		return false
 26 	},
 27 	dispatchEvent:function(event, data){
 28 		if ( this.events[event] ) {
 29 			var listeners = this.events[event], len = listeners.length;
 30 			while ( len-- ) {
 31 				listeners[len](data);	//callback with self
 32 			}		
 33 			return true; // return success
 34 		} else {
 35 			return false;
 36 		}
 37 	}
 38 }
