 53 	/** 
 54 		Call to set up Webgl, initialize the canvas, build matrices,
 55 		and start animate() loop
 56 		@function 
 57 		@public
 58 	*/
 59 	
 60 	LAB.app.TDLApp.prototype.begin = function()
 61 	{
 62 		console.log('begin');
 63 		//math and matrix
 64 		math = tdl.math;
 65 		fast = tdl.fast;
 66 
 67 		//setup canvas and context
 68 				
 69 		if (document.getElementById("canvas") != null){
 70 			this.canvas = document.getElementById("canvas");
 71 		} else {
 72 			console.log("no canvas in document, generating")
 73 			this.canvas = document.createElement( 'canvas' );
 74 			this.canvas.width 	= window.innerWidth;
 75 			this.canvas.height	= window.innerHeight;
 76 			
 77 			if (document.body)
 78 				document.body.appendChild( this.canvas );
 79 			else
 80 				return;
 81 		}
 82 		
 83 		g_fpsTimer = new tdl.fps.FPSTimer();
 84 
 85 		gl = tdl.webGL.setupWebgl(this.canvas);
 86 		if (!gl) {
 87 			console.log("ERROR SETTING UP gl");
 88 			return false;
 89 		}
 90 		if (g_debug) {
 91 			gl = tdl.webgl.makeDebugContext(gl, undefined, LogglCall);
 92 		}
 93 
 94 		// util function stored in LabUtils
 95 		// !!!! needs to be refactored
 96 		labSetMatrices();
 97 		this.setup();
 98 		this.animate();
 99 	}

101 /************************************************************
102 	UPDATING: OVERRIDE PREUPDATE TO PREP MATRICES
103 ************************************************************/
104 
105 	/** 
106 		Calculates elapsed + delta time
107 		@function 
108 		@private
109 	*/
110 
111 	LAB.app.TDLApp.prototype.preupdate = function(){
112 		var now = (new Date()).getTime();
113 		if(this.lastTime == 0.0) {
114  			this.deltaTime = 0.0;
115 	   	} else {
116 	      this.deltaTime = this.now - this.lastTime;
117 	   	}
118 		this.lastTime 		= this.now;
119 		this.labTimeDelta 	= this.deltaTime;
120 
121 		g_fpsTimer.update(this.deltaTime * .001);//<--- to milliseconds?
122 		if (this.fpsElem != null){
123 			this.fpsElem.innerHTML = g_fpsTimer.averageFPS;		
124 		} 
125 
126 		this.labFPS = g_fpsTimer.instantaneousFPS;
127 		this.elapsedTime += this.deltaTime;
128 	}

131 	DRAWING: OVERRIDE PREDRAW AND POSTDRAW TO SET UP gl CTX

134 	/** 
135 		Sets up Webgl drawing context, blending, projection/modelview matrices
136 		@function 
137 		@private
138 	*/
139 
140 	LAB.app.TDLApp.prototype.predraw = function(){
141 	   	// turn off logging after 1 frame.
142 	   	g_logglCalls = false;
143 	
144 		gl.clearColor( 1,1,1,1 );
145 		gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT );
146 
147 		gl.disable( gl.DEPTH_TEST );
148 		gl.enable( gl.BLEND );
149 		gl.blendFunc( gl.SRC_ALPHA , gl.ONE_MINUS_SRC_ALPHA);
150 	
151 		gl.viewport(0, 0, this.canvas.width, this.canvas.height);//fit viewport to screen
152 		
153 		labPushMatrix();
154 		
155 		// give you an option to push to orthographic (flat) projection
156 		if (this.bOrthographic){			
157 			labPushProjection();
158 			//remake perspective     
159 			fast.identity4( projectionMatrix );                          
160 			projectionMatrix = tdl.math.matrix4.orthographic(0, canvas.width, 0, canvas.height, nearClip, farClip);
161 			//remake modelview matrix
162 			fast.identity4(modelviewMatrix);
163 
164 			//move to upper left corner and scale. mimics openframeworks setup
165 			labScale(1,-1,1);
166 			labTranslate(0, -canvas.height, 0);
167 		}
168 	};
169

171 		Pops matrices
172 		@function 
173 		@private
174 	*/
175 	
176 	LAB.app.TDLApp.prototype.postdraw = function(){
177 		if (this.bOrthographic) labPopProjection();
178 		labPopMatrix();
179 	}
