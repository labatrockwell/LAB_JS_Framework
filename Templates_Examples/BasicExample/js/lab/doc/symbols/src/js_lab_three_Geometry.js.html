 27 LAB.three.Geometry.prototype.loadLabModel = function( model ){
 28    //vertex positions
 29    if( model.positions.length > 0 ){
 30       this.vertices = [];
 31       for(var i=0; i<model.positions.length; i++){
 32          this.vertices[i] = new THREE.Vertex( new THREE.Vector3(model.positions[i][0],
 33                                                              model.positions[i][1],
 34                                                              model.positions[i][2]));
 35       }
 36    }
 37    
 38    //vertex normals. these aren't loading to the three mesh yet. need to look into how three gets normals to the shader. I think its face->Vertex->Normals meaning seperate normals for each face vertex. not shared between faces
 39    if( model.normals.length ){// we should switch the exporter to write vertexNormals & faceNormals
 40       for(var i=0; i<model.normals.length; i++){
 41          this.vertexNormals[i] = new THREE.Vector3(model.normals[i][0],
 42                                                    model.normals[i][1],
 43                                                    model.normals[i][2]);
 44       }
 45    }
 46    
 47    
 48    for(var i=0; i<model.indices.length; i++ ){
 49       var fi = model.indices[i];
 50       if(fi.length == 3 ){
 51          this.faces.push( new THREE.Face3(fi[0],fi[1],fi[2] ) );
 52          
 53          if( model.texCoords.length ){
 54             this.faceVertexUvs[0][i] = [new THREE.UV( model.texCoords[ fi[0] ][0], model.texCoords[ fi[0] ][1]),
 55                                         new THREE.UV( model.texCoords[ fi[1] ][0], model.texCoords[ fi[1] ][1]),
 56                                         new THREE.UV( model.texCoords[ fi[2] ][0], model.texCoords[ fi[2] ][1])];
 57          }
 58       }
 59       else if( fi.length == 4){
 60          this.faces.push( new THREE.Face4(fi[0],fi[1],fi[2],fi[3] ) );
 61          
 62          if( model.texCoords.length ){
 63             this.faceVertexUvs[0][i] = [new THREE.UV( model.texCoords[ fi[0] ][0], model.texCoords[ fi[0] ][1]),
 64                                         new THREE.UV( model.texCoords[ fi[1] ][0], model.texCoords[ fi[1] ][1]),
 65                                         new THREE.UV( model.texCoords[ fi[2] ][0], model.texCoords[ fi[2] ][1]),
 66                                         new THREE.UV( model.texCoords[ fi[3] ][0], model.texCoords[ fi[3] ][1])];
 67          }
 68       }
 69    }

108 LAB.three.Geometry.prototype.getPointOnFace = function( faceIndex, u, v ){
109    
110    var f = this.faces[Math.min( faceIndex, this.faces.length - 1)];
111    var fPos = new THREE.Vector3();
112    
113    if( f instanceof THREE.Face3){  
114       /*
115       http://en.wikipedia.org/wiki/Barycentric_coordinate_system_(mathematics)
116       http://www.exaflop.org/docs/cgafaq/cga6.html 
117       If u+v>1, replace a by 1-u, v by 1-v. Let w=1-u-v. Then a*u + b*v + c*w is uniformly distributed 
118       in triangle ABC: the reflection step u=1-u; v=1-v gives a point (u,v) uniformly distributed in 
119       the triangle (0,0)(1,0)(0,1), which is then mapped affinely to ABC. Now you have barycentric 
120       coordinates a,b,c. Compute your point P = a*u + b*v + c*w.
121        */
122       
123       if(u+v > 1){
124          u = 1 - u;
125          v = 1 - v;
126       }
127       var w = 1-u-v;
128       var a = this.vertices[f.a].position;
129       var b = this.vertices[f.b].position;
130       var c = this.vertices[f.c].position;
131       fPos.set(a.x*u + b.x*v + c.x * w,
132                a.y*u + b.y*v + c.y * w,
133                a.z*u + b.z*v + c.z * w );
134       
135    }
136    else if(f instanceof THREE.Face4){
137       //untested
138       var mu = 1.0 - u;
139       var mv = 1.0 - v;
140       var a = this.vertices[f.a].position;
141       var b = this.vertices[f.b].position;
142       var c = this.vertices[f.c].position;
143       var d = this.vertices[f.d].position;
144       fPos.set((a.x*mu + b.x*u)*mv + (d.x*mu + c.x*u)*v,
145                (a.y*mu + b.y*u)*mv + (d.y*mu + c.y*u)*v,
146                (a.z*mu + b.z*u)*mv + (d.z*mu + c.z*u)*v);
147    }
148    
149    return fPos;
150 };

152 LAB.three.Geometry.prototype.getSmoothedNormalOnFace = function( faceIndex, u, v ){
153    var f = this.faces[Math.min( faceIndex, this.faces.length - 1)];
154    var fNorm = new THREE.Vector3();
155    
156    if( f instanceof THREE.Face3){  
157       
158       if(u+v > 1){
159          u = 1 - u;
160          v = 1 - v;
161       }
162       var w = 1-u-v;
163       var a = this.getVertexNormal( f.a );
164       var b = this.getVertexNormal( f.b );
165       var c = this.getVertexNormal( f.c );
166       fNorm.set(a.x*u + b.x*v + c.x * w,
167                a.y*u + b.y*v + c.y * w,
168                a.z*u + b.z*v + c.z * w );
169       
170    }
171    else if(f instanceof THREE.Face4){
172       //untested
173       var mu = 1.0 - u;
174       var mv = 1.0 - v;
175       var a = this.getVertexNormal( f.a );
176       var b = this.getVertexNormal( f.b );
177       var c = this.getVertexNormal( f.c );
178       var d = this.getVertexNormal( f.d );

193 LAB.three.Geometry.prototype.findEdges = function(){
194    
195    //create edges 
196    for(var i=0; i<this.faces.length; i++){
197       f = this.faces[i];
198       if ( f instanceof THREE.Face3 ){
199          var e = new THREE.Edge(this.vertices[f.a], this.vertices[f.b], f.a, f.b); 
200          e.faces.push( f );
201          e.faceIndices.push( i );
202          this.edges.push( e );
203          
204          e = new THREE.Edge(this.vertices[f.b], this.vertices[f.c], f.b, f.c ); 
205          e.faces.push( f );
206          e.faceIndices.push( i );
207          this.edges.push( e );
208          
209          e = new THREE.Edge(this.vertices[f.c], this.vertices[f.a], f.c, f.a); 
210          e.faces.push( f );
211          e.faceIndices.push( i );
212          this.edges.push( e );
213       }
214       else if (f instanceof THREE.Face4 ){
215          var e = new THREE.Edge(this.vertices[f.a], this.vertices[f.b], f.a, f.b); 
216          e.faces.push( f );
217          e.faceIndices.push( i );
218          this.edges.push( e );
219          
220          e = new THREE.Edge(this.vertices[f.b], this.vertices[f.c], f.b, f.c ); 
221          e.faces.push( f );
222          e.faceIndices.push( i );
223          this.edges.push( e );
224          
225          e = new THREE.Edge(this.vertices[f.c], this.vertices[f.d], f.c, f.d); 
226          e.faces.push( f );
227          e.faceIndices.push( i );
228          this.edges.push( e );
229          
230          e = new THREE.Edge(this.vertices[f.d], this.vertices[f.a], f.d, f.a); 
231          e.faces.push( f );
232          e.faceIndices.push( i );
233          this.edges.push( e );
234       }
235    }
236    
237    //merge duplicate edges
238    var fi, fr;//face index and face reference
239    function mergeEdges( _e1, _e2 ){
240       for(var i=_e2.faces.length-1; i >= 0; i--){
241          fr = _e2.faces.pop();
242          fi = _e2.faceIndices.pop();
243          
244          for(var j=0; j<_e1.faces.length;j++){
245             if( fi != _e1.faceIndices[j]){
246                _e1.faces.push( fr );
247                _e1.faceIndices.push( fi );
248             }
249          }
250       }
251    }
252    
253    var e1, e2;
254    for(var i=0; i<this.edges.length; i++){
255       e1 = this.edges[i];
256       for(var j=i+1; j<this.edges.length; j++){
257          e2 = this.edges[j];
258          if( e1.vertexIndices[0] == e2.vertexIndices[0] && e1.vertexIndices[1] == e2.vertexIndices[1]){
259             mergeEdges( e1, e2 );
260          }
261          else if(e1.vertexIndices[1] == e2.vertexIndices[0] && e1.vertexIndices[0] == e2.vertexIndices[1]){
262             mergeEdges( e1, e2 );
263          }
264       }
265    }
266    
267    for(var i=this.edges.length-1; i>=0; i--){
268       if(this.edges[i].faces.length == 0){
269          this.edges.splice( i, 1 );
270       }
271    }
272 //   labLog( this.edges.length );
273 //   labLog( this.edges );
274 };

276 LAB.three.Geometry.prototype.findVertexEdges = function(){
277    
278    //this is structured so that we can traverse to a vertex's edges using the same index as our vertex
279    if(this.edges.length == 0){
280       this.findEdges();
281    }
282    
283    //create or empty vertexEdges
284    for(var i=0; i<this.vertices.length; i++){
285       this.vertexEdges[i] = [];
286    }
287    
288    //add edges to vertexEdges
289    for(var i=0; i<this.edges.length; i++){
290       //[ edgeIndex, edgeReference ]
291       this.vertexEdges[ this.edges[i].vertexIndices[0] ].push( [ i, this.edges[i] ]);
292       this.vertexEdges[ this.edges[i].vertexIndices[1] ].push( [ i, this.edges[i] ]);
293    }
294    //   labLog( this.vertexEdges.length );
295    //   labLog( this.vertexEdges );
296 };

298 LAB.three.Geometry.prototype.findVertexFaces = function(){
299 
300    //create or empty vertexEdges
301    for(var i=0; i<this.vertices.length; i++){
302       this.vertexFaces[i] = [];
303    }
304    
305    //add faces to vertexFaces
306    var f;
307    for(var i=0; i<this.faces.length; i++){
308       f = this.faces[i];
309       if(f instanceof THREE.Face3 ){
310          this.vertexFaces[ f.a ].push( [ i, f ] );
311          this.vertexFaces[ f.b ].push( [ i, f ] );
312          this.vertexFaces[ f.c ].push( [ i, f ] );
313       }
314       else if(f instanceof THREE.Face4 ){
315          this.vertexFaces[ f.a ].push( [ i, f ] );
316          this.vertexFaces[ f.b ].push( [ i, f ] );
317          this.vertexFaces[ f.c ].push( [ i, f ] );
318          this.vertexFaces[ f.d ].push( [ i, f ] );
319       }
320    }
321 //   labLog( this.vertexFaces.length );
322 //   labLog( this.vertexFaces );

326 LAB.three.Geometry.prototype.findFaceEdges = function(){
327    
328    //this is structured so that we can traverse to a vertex's edges using the same index as our vertex
329    if(this.edges.length == 0){
330       this.findEdges();
331    }
332    
333    //create or empty faceEdges // is this the best/right way to do this?
334    for(var i=0; i<this.faces.length; i++){
335       this.faceEdges[i] = [];
336    }
337    
338    //add edges to vertexEdges
339    for(var i=0; i<this.edges.length; i++){
340       
341       for(var j=0; j<this.edges[i].faceIndices.length; j++) // non-manifold geometry can have n faces per edge
342       {
343          this.faceEdges[ this.edges[i].faceIndices[j] ].push( [ i, this.edges[i] ]);//[ edgeIndex, edgeReference ]
344       }
345    }

365 LAB.three.Geometry.prototype.calcVertexNormal = function( vertexIndex ){
366    //not as easy as it sounds in three o_O
367    //for now we'll just calculate it using vertexFaces and store them in this.vertexNormals...
368    if( !this.vertexNormals[vertexIndex] ){
369       this.vertexNormals[vertexIndex] = new THREE.Vector3();
370    }
371    
372    var vf = this.vertexFaces[ vertexIndex ];
373    for(var i=0; i<vf.length; i++){
374       this.vertexNormals[vertexIndex].addSelf( vf[i][1].normal );
375    }
376    
377    return this.vertexNormals[vertexIndex].normalize();
378 };
