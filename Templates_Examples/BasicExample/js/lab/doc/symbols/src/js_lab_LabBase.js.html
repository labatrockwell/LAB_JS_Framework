  5 // reference to global context, in most cases 'window'.

  8 // require based on: http://closure-library.googlecode.com/svn/trunk/closure/goog/base.js
  9 
 10 /*
 11  * helper (for javascript importing within javascript).
 12  */
 13 
 14 var included = {}
 15 
 16 /**
 17 @function
 18 @private
 19 */
 20 LAB.writeScriptTag_ = function(src) {
 21 	var doc = LAB.global.document;
 22 	doc.write(
 23 		'<script type="text/javascript" src="' + src + '"></' + 'script>');
 24 	included[src] = true;
 25 	return true;
 26 }
 27 
 28 /**
 29 	Use this to load script files! (only script files right now)
 30 	@function
 31 	@public
 32 */
 33 LAB.require = function(src) {
 34 	src in included ? console.log("already included") : LAB.writeScriptTag_(src);
 35 }
