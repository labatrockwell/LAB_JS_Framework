  5 /** @namespace LAB */
  6 LAB.BaseApp = function()
  7 {
  8 		
  9 	LAB.self 	= this;
 10 	LAB.EventDispatcher.call( this );
 11 	
 12 	//utils
 13 	this.mouse	= {x:0, y:0, bDown:false};
 14 	this.startTime 		= Date.now();
 15 	this.elapsedTime 	= 0;
 16 }
 17 
 18 LAB.BaseApp.prototype				= new LAB.EventDispatcher();
 19 LAB.BaseApp.prototype.constructor 	= LAB.BaseApp;
 20 LAB.BaseApp.prototype.supr 			= LAB.EventDispatcher.prototype;
 21 
 22 LAB.BaseApp.prototype.begin = function(){
 23 	console.log("base app set up");
 24 	this.setup();
 25 	this.animate();
 26 };
 27 
 28 // suggestions for overriding: updating + rendering
 29 
 30 LAB.BaseApp.prototype.setup 	= function(){};
 31 LAB.BaseApp.prototype.preupdate = function(){};
 32 LAB.BaseApp.prototype.update	= function(){};
 33 LAB.BaseApp.prototype.predraw	= function(){};	// e.g. setup screen
 34 LAB.BaseApp.prototype.draw		= function(){};	// override this in your main app
 35 LAB.BaseApp.prototype.postdraw 	= function(){}; // e.g. teardown screen
 36 
 37 // starts off animation + sets up update/draw loop
 38 LAB.BaseApp.prototype.animate	= function(time){
 39 	// update time
 40 	if (time === undefined){
 41 	} else {
 42 		LAB.self.elapsedTime = time - LAB.self.startTime;
 43 	}
 44 	
 45 	requestAnimationFrame(LAB.self.animate, this);
 46 			
 47 	LAB.self.preupdate();
 48    	LAB.self.update();
 49 	LAB.self.predraw();
 50 	LAB.self.draw();
 51 	LAB.self.postdraw();
 52 }

 58 	// turn events on + off
 59 
 60 	LAB.BaseApp.prototype.registerMouseEvents = function(){
 61 		window.addEventListener("mousemove", LAB.self._onMouseMoved);
 62 		window.addEventListener("mousedown", LAB.self._onMousePressed);
 63 		window.addEventListener("mouseup", LAB.self._onMouseReleased);
 64 	}
 65 
 66 	LAB.BaseApp.prototype.unregisterMouseEvents = function()
 67 	{
 68 		window.removeEventListener("mousemove", LAB.self._onMouseMoved);
 69 		window.removeEventListener("mousedown", LAB.self._onMousePressed);
 70 		window.removeEventListener("mouseup", LAB.self._onMouseReleased);
 71 	}

 73 	// OVERRIDE THESE FUNCTIONS IN YOUR MAIN APP!
 74 	
 75 	LAB.BaseApp.prototype.onMouseMoved		= function (x,y){};
 76 	LAB.BaseApp.prototype.onMousePressed	= function (x,y){};
 77 	LAB.BaseApp.prototype.onMouseReleased	= function (x,y){};
 78 
 79 	// these could maybe be their jquery equivalents?
 80 	// note: these can be used 1 of 2 ways: 
 81 	//	1 - overridden: catch the raw event in your app that inherits BaseApp
 82 	// 	2 - event dispatchers: add event listeners to any or all of the mouse events!
 83 
 84 	LAB.BaseApp.prototype._onMouseMoved		= function( event )
 85 	{	 
 86 		// if the mouse is down, call dragged instead of moved
 87 		if (LAB.self.mouse.bDown){
 88 			LAB.self.onMouseDragged(event);
 89 			return;
 90 		}
 91 		LAB.self.mouse.x = event.clientX; 
 92 		LAB.self.mouse.y = event.clientY;
 93 		LAB.self.dispatchEvent("onMouseMoved", LAB.self.mouse);
 94 		LAB.self.onMouseMoved(LAB.self.mouse.x, LAB.self.mouse.y);
 95 	}
 96 
 97 	LAB.BaseApp.prototype._onMousePressed	= function( event )
 98 	{
 99 		LAB.self.mouse.x = event.clientX; 
100 		LAB.self.mouse.y = event.clientY;
101 		LAB.self.mouse.bDown = true;
102 		LAB.self.dispatchEvent("onMousePressed", LAB.self.mouse);
103 		LAB.self.onMousePressed(LAB.self.mouse.x, LAB.self.mouse.y);
104 	}
105 
106 	LAB.BaseApp.prototype._onMouseReleased	= function( event )
107 	{
108 		LAB.self.mouse.x = event.clientX; 
109 		LAB.self.mouse.y = event.clientY;
110 		LAB.self.mouse.bDown = false;
111 		LAB.self.dispatchEvent("onMouseReleased", LAB.self.mouse);
112 		LAB.self.onMouseReleased(LAB.self.mouse.x, LAB.self.mouse.y);
113 	}
114 
115 	LAB.BaseApp.prototype._onMouseDragged	= function( event )
116 	{
117 		LAB.self.mouse.x = event.clientX; 
118 		LAB.self.mouse.y = event.clientY;
119 		LAB.self.dispatchEvent("onMouseDragged", LAB.self.mouse);
120 		LAB.self.onMouseDragged(LAB.self.mouse.x, LAB.self.mouse.y);
121 	}
