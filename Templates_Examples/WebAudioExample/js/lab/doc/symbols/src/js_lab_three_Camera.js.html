 35 /**
 36 	project a THREE.Vector3 from world to screen coords
 37 	@function
 38 	@public
 39  */
 40 LAB.three.Camera.prototype.projectToScreen = function( worldPos ){
 41    //adaptded from https://github.com/mrdoob/three.js/issues/78
 42    var pos = worldPos.clone();
 43    projScreenMat = new THREE.Matrix4();
 44    projScreenMat.multiply( this.projectionMatrix, this.matrixWorldInverse );
 45    projScreenMat.multiplyVector3( pos );
 46    
 47    return { x: ( pos.x + 1 ) * window.innerWidth / 2 ,
 48             y: window.innerHeight - ( - pos.y + 1) * window.innerHeight / 2,
 49             z: 0 };
 50 };
 51 
 52 /**
 53  project a THREE.Vector3 from screen coords to world space
 54  @function
 55  @public
 56  */
 57 LAB.three.Camera.prototype.projectToWorld = function( screenPos ){
 58    //this was helpful http://jsfiddle.net/gero3/PE4x7/25/
 59    var pos = screenPos.clone();
 60    pos.x = ( pos.x / window.innerWidth ) * 2 - 1;
 61    pos.y = ( -(window.innerHeight-pos.y) / window.innerHeight ) * 2 + 1;
 62    
 63    var projector = new THREE.Projector();
 64    projector.unprojectVector( pos, this );
 65    return pos;
 66 };
