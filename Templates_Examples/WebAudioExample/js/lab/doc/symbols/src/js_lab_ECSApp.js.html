 18 	LAB.ECSApp.prototype.setupECS 		= 	function(uuid, inport, outport)
 19 	{	
 20 		this.uuid			= uuid;
 21 		this.ecs_in_port 	= inport;
 22 		this.ecs_out_port 	= outport;
 23 
 24 		this._api = new ApplicationMessageHandler(uuid);
 25 
 26 		this._api.addEventListener("handleStartPerformance", this.handleStartPerformance);
 27 		this._api.addEventListener("handleStopPerformance", this.handleStopPerformance);
 28 		this._api.addEventListener("handleConfig", this.handleConfig);
 29 		this._api.addEventListener("handlePerformanceConfig", this.handlePerformanceConfig);
 30 				
 31 		this._api.connect(this.ecs_in_port, this.ecs_out_port);
 32 		this.bECSSetup = true;	
 33 	};
